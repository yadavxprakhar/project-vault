## 🚀 AST Analysis Example
```python
# Using Python's ast module
import ast

code = """
def calculate_sum(a, b):
    result = a + b
    return result
"""

tree = ast.parse(code)

class CodeAnalyzer(ast.NodeVisitor):
    def __init__(self):
        self.issues = []
    
    def visit_FunctionDef(self, node):
        # Check function complexity
        if len(node.body) > 20:
            self.issues.append(f"Function '{node.name}' is too complex")
        
        # Check for missing docstring
        if not ast.get_docstring(node):
            self.issues.append(f"Function '{node.name}' missing docstring")
        
        self.generic_visit(node)
    
    def visit_Name(self, node):
        # Check variable naming conventions
        if node.id.isupper() and not node.id.startswith('_'):
            self.issues.append(f"Variable '{node.id}' should use snake_case")
        
        self.generic_visit(node)

analyzer = CodeAnalyzer()
analyzer.visit(tree)
print(analyzer.issues)

from openai import OpenAI

client = OpenAI(api_key="your-api-key")

def review_code_with_gpt(code: str, language: str) -> dict:
    prompt = f"""
    Review the following {language} code and provide:
    1. Potential bugs or issues
    2. Code quality improvements
    3. Security vulnerabilities
    4. Performance optimizations
    
    Code:
    ```{language}
    {code}
    ```
    
    Format your response as JSON with keys: bugs, improvements, security, performance
    """
    
    response = client.chat.completions.create(
        model="gpt-4-turbo-preview",
        messages=[{"role": "user", "content": prompt}],
        response_format={"type": "json_object"}
    )
    
    return response.choices[0].message.content

### 📊 GitHub Integration Example (JavaScript)

// Using Octokit to comment on PR
const { Octokit } = require('@octokit/rest');

const octokit = new Octokit({
  auth: process.env.GITHUB_TOKEN
});

async function addReviewComment(owner, repo, pullNumber, issues) {
  const comments = issues.map(issue => ({
    path: issue.file,
    position: issue.line,
    body: `🤖 **AI Code Review**\n\n${issue.message}\n\n**Suggestion:**\n\`\`\`${issue.language}\n${issue.suggestion}\n\`\`\``
  }));

  await octokit.pulls.createReview({
    owner,
    repo,
    pull_number: pullNumber,
    event: 'COMMENT',
    comments
  });
}

### 🔍 Tree-sitter Example (JavaScript)

const Parser = require('tree-sitter');
const JavaScript = require('tree-sitter-javascript');

const parser = new Parser();
parser.setLanguage(JavaScript);

const code = `
function processData(data) {
  if (data == null) {  // Should use ===
    return [];
  }
  return data.map(item => item.value);
}
`;

const tree = parser.parse(code);

function findIssues(node) {
  const issues = [];
  
  if (node.type === 'binary_expression' && node.text.includes('==')) {
    issues.push({
      type: 'style',
      message: 'Use === instead of ==',
      line: node.startPosition.row + 1
    });
  }
  
  for (const child of node.children) {
    issues.push(...findIssues(child));
  }
  
  return issues;
}

const issues = findIssues(tree.rootNode);

### 📈 Complexity Metrics Example (Python)

# Calculate cyclomatic complexity
def calculate_complexity(tree):
    complexity = 1  # Base complexity
    
    for node in ast.walk(tree):
        # Add 1 for each decision point
        if isinstance(node, (ast.If, ast.While, ast.For, ast.ExceptHandler)):
            complexity += 1
        elif isinstance(node, ast.BoolOp):
            complexity += len(node.values) - 1
    
    return complexity

# Usage
code = """
def complex_function(x):
    if x > 0:
        if x < 10:
            return "small"
        else:
            return "large"
    else:
        return "negative"
"""

tree = ast.parse(code)
complexity = calculate_complexity(tree)
print(f"Cyclomatic Complexity: {complexity}")
