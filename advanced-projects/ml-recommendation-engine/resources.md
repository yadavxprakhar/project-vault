## 🧠 Neural Collaborative Filtering (PyTorch)
```Python

import torch
import torch.nn as nn

class NCF(nn.Module):
    def __init__(self, num_users, num_items, embedding_dim=64, hidden_layers=[128, 64, 32]):
        super(NCF, self).__init__()
        
        # User and item embeddings
        self.user_embedding = nn.Embedding(num_users, embedding_dim)
        self.item_embedding = nn.Embedding(num_items, embedding_dim)
        
        # MLP layers
        layers = []
        input_dim = embedding_dim * 2
        for hidden_dim in hidden_layers:
            layers.append(nn.Linear(input_dim, hidden_dim))
            layers.append(nn.ReLU())
            layers.append(nn.Dropout(0.2))
            input_dim = hidden_dim
        
        self.mlp = nn.Sequential(*layers)
        self.output = nn.Linear(hidden_layers[-1], 1)
        self.sigmoid = nn.Sigmoid()
    
    def forward(self, user_ids, item_ids):
        user_embed = self.user_embedding(user_ids)
        item_embed = self.item_embedding(item_ids)
        
        # Concatenate embeddings
        x = torch.cat([user_embed, item_embed], dim=-1)
        
        # Pass through MLP
        x = self.mlp(x)
        x = self.output(x)
        
        return self.sigmoid(x)

# Training
model = NCF(num_users=10000, num_items=5000)
optimizer = torch.optim.Adam(model.parameters(), lr=0.001)

🎯 Content-Based Filtering (TF-IDF + Cosine Similarity)
```Python

from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.metrics.pairwise import cosine_similarity
import pandas as pd

# Item metadata (e.g., movie descriptions)
items_df = pd.DataFrame({
    'item_id': [1, 2, 3, 4],
    'description': [
        'Action thriller with explosions',
        'Romantic comedy about love',
        'Action adventure with cars',
        'Romantic drama with emotions'
    ]
})

# Create TF-IDF matrix
tfidf = TfidfVectorizer(stop_words='english')
tfidf_matrix = tfidf.fit_transform(items_df['description'])

# Compute similarity
cosine_sim = cosine_similarity(tfidf_matrix, tfidf_matrix)

def get_recommendations(item_id, top_n=5):
    # Get similarity scores for this item
    idx = items_df[items_df['item_id'] == item_id].index[0]
    sim_scores = list(enumerate(cosine_sim[idx]))
    
    # Sort by similarity
    sim_scores = sorted(sim_scores, key=lambda x: x[1], reverse=True)
    
    # Get top N similar items (excluding itself)
    sim_scores = sim_scores[1:top_n+1]
    item_indices = [i[0] for i in sim_scores]
    
    return items_df.iloc[item_indices]['item_id'].tolist()

# Get recommendations for item 1
recommendations = get_recommendations(1, top_n=3)
print(f"Similar items to item 1: {recommendations}")
```

## 🔄 Hybrid Recommendation System
```Python

class HybridRecommender:
    def __init__(self, collaborative_model, content_model, alpha=0.7):
        self.collaborative_model = collaborative_model
        self.content_model = content_model
        self.alpha = alpha  # Weight for collaborative filtering
    
    def predict(self, user_id, item_id):
        # Get collaborative filtering score
        cf_score = self.collaborative_model.predict(user_id, item_id)
        
        # Get content-based score
        cb_score = self.content_model.get_similarity_score(user_id, item_id)
        
        # Hybrid score (weighted combination)
        hybrid_score = self.alpha * cf_score + (1 - self.alpha) * cb_score
        
        return hybrid_score
    
    def recommend(self, user_id, n=10):
        # Get candidate items
        candidates = self.get_candidates(user_id)
        
        # Score all candidates
        scores = [(item_id, self.predict(user_id, item_id)) 
                  for item_id in candidates]
        
        # Sort and return top N
        scores.sort(key=lambda x: x[1], reverse=True)
        return [item_id for item_id, score in scores[:n]]

class ColdStartHandler:

        self.content_model = content_model
    
    def recommend_for_new_user(self, user_profile=None, n=10):

            # Use content-based filtering with user profile

            # Return popular items

## 🧪 A/B Testing Framework
```Python

import random
from scipy import stats

class ABTest:
    def __init__(self, model_a, model_b, traffic_split=0.5):
        self.model_a = model_a
        self.model_b = model_b
        self.traffic_split = traffic_split
        self.results_a = []
        self.results_b = []
    
    def get_recommendation(self, user_id):
        # Random assignment
        if random.random() < self.traffic_split:
            variant = 'A'
            recommendations = self.model_a.recommend(user_id)
        else:
            variant = 'B'
            recommendations = self.model_b.recommend(user_id)
        
        return recommendations, variant
    
    def record_feedback(self, variant, metric_value):
        if variant == 'A':
            self.results_a.append(metric_value)
        else:
            self.results_b.append(metric_value)
    
    def analyze_results(self):
        # Perform t-test
        t_stat, p_value = stats.ttest_ind(self.results_a, self.results_b)
        
        mean_a = np.mean(self.results_a)
        mean_b = np.mean(self.results_b)
        
        return {
            'model_a_mean': mean_a,
            'model_b_mean': mean_b,
            'improvement': (mean_b - mean_a) / mean_a * 100,
            'p_value': p_value,
            'significant': p_value < 0.05
        }
```
